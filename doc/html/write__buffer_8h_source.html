275 {
276  if (!cio_write_buffer_queue_empty(list)) {
277  struct cio_write_buffer *new_last = list->prev;

313 static inline void cio_write_buffer_split_and_append(struct cio_write_buffer *to_head, struct cio_write_buffer *from_head, struct cio_write_buffer *wbe)
314 {
315  if (cio_unlikely(cio_write_buffer_queue_empty(from_head))) {
316  return;
317  }
318 
319  if (cio_unlikely(from_head == wbe)) {
320  wbe = from_head->next;
321  }
322 
323  struct cio_write_buffer *to_last = to_head->prev;
324  struct cio_write_buffer *from_last = from_head->prev;
325  size_t tail_length = 0;
326  size_t tail_qlen = 0;
327 
328  const struct cio_write_buffer *element = wbe;
329  while (element != from_head) {
330  tail_length += element->data.element.length;
331  tail_qlen++;
332  element = element->next;
333  }
334 
335  struct cio_write_buffer *wbe_prev = wbe->prev;
336  wbe_prev->next = from_head;
337  from_head->prev = wbe_prev;
338  from_head->data.head.q_len -= tail_qlen;
339  from_head->data.head.total_length -= tail_length;
340 
341  from_last->next = to_head;
342  to_head->prev = from_last;
343  to_last->next = wbe;
344  wbe->prev = to_last;
345  to_head->data.head.q_len += tail_qlen;
346  to_head->data.head.total_length += tail_length;
347 }
