87 static inline enum cio_error cio_read_buffer_init(struct cio_read_buffer *rb, void *data, size_t size)
88 {
89  if (cio_unlikely((rb == NULL) || (data == NULL) || (size == 0))) {
90  return CIO_INVALID_ARGUMENT;
91  }
92 
93  rb->data = data;
94  rb->end = (uint8_t *)data + size;
95  rb->fetch_ptr = data;
96  rb->add_ptr = data;
97 
98  return CIO_SUCCESS;
99 }
